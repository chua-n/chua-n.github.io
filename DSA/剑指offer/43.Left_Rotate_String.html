#include <string>
using namespace std;

class Solution {
public:
    string LeftRotateString(string str, int n) {
        int length = str.size();
        if (length == 0) return str;

        n %= length;
        auto copy = str.substr(0, n);
        for (int i = 0; i < length; ++i) {
            str[i] = i + n < length ? str[i + n] : copy[i + n - length];
        }
        return str;
    }
};
